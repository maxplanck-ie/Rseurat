Annotate query and filter identities

pbmc@meta.data$SingleR_labels <- ref.labels$pruned.labels
pbmc@meta.data$SingleR_score <- apply(ref.labels$scores,
    1, function(X) X[which.max(X)])
pbmc[["SingleR_labels"]][pbmc[["SingleR_score"]] <
    0.5] <- NA

Seurat Label Transfer: transfer labels

pbmc.anchors <- FindTransferAnchors(reference = pbmc_ref.seurat,
    query = pbmc, dims = 1:30, reference.reduction = "pca")
predictions <- TransferData(anchorset = pbmc.anchors,
    refdata = pbmc_ref.seurat$label, dims = 1:30)
pbmc <- AddMetaData(pbmc, metadata = predictions)
